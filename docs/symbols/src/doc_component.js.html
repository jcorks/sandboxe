1 /**
  2  * @class component
  3  * @description
  4  * Extends the functionality of an {@link entity} as a swappable object. </br></br>
  5  *
  6  * To work, all components have hosts; when a host updates, all components do as well.
  7  * Like {@link entity}, a component can have a Draw and a Step function with an overloadable
  8  * OnStep / OnDraw. In addition, all component's also have event handling capabilities.
  9  * Setable with strings, it is possible to maintain and assign functions to run on 
 10  * certain events and even define your own events for custom components.</br></br>
 11  *  
 12  * When making your own components, you will typically also want to include 
 13  * your own custom events. installEvent() is the function that adds 
 14  * a new recognized event with the given name. Once installed, emitEvent() should be called 
 15  * at the appropriate time user functions should be run. </br></br>
 16  *
 17  * Built-in components, link {@link clock}, have their installed events marked in their 
 18  * class descriptions.
 19  * @example 
 20  
 21  
 22  *
 23  */ 
 24 function component() {
 25 
 26     /**
 27      * Does the naturally defined step iteration. This invokes the "onStep" property method
 28      * if defined. For built-in components, this may trigger native code.
 29      * For hosts that are part of the engine hierarchy, step() is called for you when 
 30      * the entity step()s.
 31      */
 32     this.step = function(){};
 33 
 34     /**
 35      * Does the naturally defined draw iteration. This invokes the "onDraw" property method
 36      * if defined. For built-in components, this may trigger native code.
 37      * For hosts that are part of the engine hierarchy, draw() is called for you when 
 38      * the entity draw()s.
 39      */
 40     this.draw = function(){};
 41 
 42     /**
 43      * Adds a new event to be recognized by the EventSystem.
 44      *
 45      * if mainHandler is undefined, the event is still added, but has no default
 46      * handler is set. The default handler is always guaranteed to run last for the event.
 47      * @param {String} eventName The name of the new event to install.
 48      * @param {Function} mainHandler (optional) The default handler for this event. Always called first before any hooks. See {@link component.installHandler} for parameters passed to handler function objects.
 49      */
 50     this.installEvent = function(eventName, mainHandler){};
 51 
 52     /**
 53      * Removes the event from the component.
 54      * @param {String} event Event to remove.
 55      */
 56     this.uninstallEvent = function(event){};
 57 
 58     /**
 59      * Triggers the specified event for this component.
 60      *
 61      * Returns whether the event was allowed to propogate all the
 62      * way to the main handler (in other words, lets you know whether
 63      * all the handlers allowed the event to happen)
 64      * Source is passed to the handler and is usually the source of the event (but does not have to be)
 65      * @param {String} eventName Name of the event. This should match the name that was installed.
 66      * @param {entity} source (Optional) entity that indicates the source of the event. For example, in a collision, this may be the object collided with.
 67      */
 68     this.emitEvent = function(eventName, source){};
 69 
 70     /**
 71      * Returns whether the given event is installed.
 72      * @param {String} eventName The event to query.
 73      */
 74     this.canHandleEvent = function(eventName){};
 75 
 76     /**
 77      * Adds a hook to the event. 
 78      *
 79      * A hook happens at the end of a given event after all the
 80      * handlers have been run. Hooks occur regardless of event handler propogation.
 81      * Hook function signatures contain the following parameters: </br></br>
 82      * <pre>
 83      * hook(component, componentHost, eventSource)
 84      * </pre></br></br>
 85      * Where component is the this component, componentHost is the host entity of this component 
 86      * , and eventSource is the entity given from the emitEvent() call, if any.
 87      * @param {String} eventName Name of the installed event.
 88      * @param {Function} hook The hook to add. See class description for function signature.
 89      */
 90     this.installHook = function(eventName, hook){};
 91 
 92     /**
 93      * Removes the given hook.
 94      * @param {String} eventName
 95      */
 96     this.uninstallHook = function(eventName){};
 97 
 98     /**
 99      * Adds a handler to an event. 
100      *
101      * Handlers that are added are run in LIFO order
102      * and their return values dictate whether the event should propogate.
103      * The last handler run for an event is always the main handler of the event. </br>
104      * Handler function signatures contain the following parameters: </br></br>
105      * <pre>
106      * {boolean} handler(component, componentHost, eventSource)
107      * </pre></br></br>
108      * Where component is the this component, componentHost is the host entity of this component 
109      * , and eventSource is the entity given from the emitEvent() call, if any. The return value 
110      * is a boolean determining if the event should continue propogating. If no return value is given, 
111      * the value is assumed true, meaning the event should propogate.
112      * @param {String} eventName Name of the installed event to add a handler to
113      * @param {Function} handler Function to add as the handler.
114      */
115     this.installHandler = function(eventName, handler){};
116 
117     /**
118      * Removes a previously installed handler.
119      * @param {String} eventName
120      */
121     this.uninstallHandler = function(eventName){};
122 
123     /**
124      * Returns a list of all events installed that can be processed.
125      * @returns {Array} An array of event name strings.
126      */
127     this.getKnownEvents = function(){};
128 
129 
130 
131 
132     /**
133      * Function called on the step iteration of the component.
134      * @type {Function}
135      */
136     this.onStep = {};
137 
138     /**
139     * Function called on the draw iteration of the component.
140      * @type {Function}
141      */
142     this.onDraw = {};
143     
144     /**
145      * Function called when the component is attached to a host.
146      * @type {Function}
147      */
148     this.onAttach = {};
149 
150 
151     /**
152      * Identifying tag of the component. This is set at the creation of the component and 
153      * is readonly. The tag can be used in access functions such as {@link entity.queryComponent}.
154      * @type {String}
155      */
156     this.tag = 0;
157 
158     /**
159      * Human-readable information on this component viable in the inspector/debugger. 
160      * For built-in components, this usually contains stats and useful info regarding the current 
161      * state of the component.
162      * @type {String}
163      */
164     this.info = 0;
165 
166     /**
167      * Whether this component should have its step() function called automatically by the engine. 
168      * The default is true.
169      * @type {Boolean}
170      */
171     this.isStepping = 0;
172 
173     /**
174      * Whether this component should have its draw() function called automatically by the engine. 
175      * The default is true.
176      * @type {Boolean}
177      */
178     this.isDrawing = 0;
179 
180     /**
181      * The current host entity of the component. See {@link entity.addComponent}.
182      * @type {entity}
183      */
184     this.host = 0;
185 
186 
187 
188     return this;
189 }
